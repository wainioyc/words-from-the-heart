 //异步
 var fs = require('fs');//引入FS模块
 var dirfile = '../words-from-the-heart';//要查询文件路劲
 var xierufile = '../js.txt';//要写入的文件路劲
 function callback(err,files){
   if (err){
     console.log('读取文件失败');
   }
   if(files.length>0){
     console.log(files);
    } else{
     console.log('没有找到文件');
    
   }
 };//定义回调函数
 fs.readdir(dirfile,callback)//调用FS模块
 var files = fs.readdirSync(dirfile)
 function callbackxie(err){
   if(err){
     console.log('写入文件失败');
   }else{
     console.log('写入文件成功');
   }
 };
 fs.writeFile(xierufile,files,callbackxie);
